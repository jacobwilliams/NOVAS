    pure elemental function anmp ( a ) result(w)

    implicit none

    real(wp),intent(in) :: a
    real(wp) :: w

    w = mod(a,twopi)
    if ( abs(w) >= pi ) w = w - sign(twopi,a)

    end function anmp
